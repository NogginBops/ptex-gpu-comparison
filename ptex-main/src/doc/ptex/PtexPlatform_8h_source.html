208template <typename T>
209PTEX_INLINE T AtomicAdd(volatile T* target, T value)
210{
211 switch(sizeof(T)){
212 case 4:
213 return (T)ATOMIC_ADD32(target, value);
214 break;
215 case 8:
216 return (T)ATOMIC_ADD64(target, value);
217 break;
218 default:
219 assert(0=="Can only use 32 or 64 bit atomics");
220 return *(T*)NULL;
221 }
222}
